13     if (phoneMatches && phoneMatches[0]) {
14         result.phone = phoneMatches[0];
15     }
16

20         .trim();
21     
22     const validParts = textWithoutSpecial.split(' ').filter(part => part.trim() !== '');
23     
24     validParts.forEach(part => {
25         if (phoneRegex.test(part)) return;
26         
27         const isName = /^[\u4e00-\u9fa5a-zA-Z]+$/.test(part) && part.length >= 1 && part.length <= 10;
28         const isAddress = /[\u4e00-\u9fa50-9]/.test(part);
29         
30         if (isName && !result.name) {
31             result.name = part;
32         }
33         else if (isAddress) {
34             result.address = result.address
35                 ? `${result.address} ${part}`
36                 : part;
37         }
38     });
39     
40     if (!result.name) {

42         if (nameMatch) {
